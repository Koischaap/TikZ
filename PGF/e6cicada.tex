\documentclass{standalone}
\usepackage{pgfplots}
\pgfplotsset{width=10cm, compat=1.16}

\begin{document}

\begin{tikzpicture}
\begin{axis}
    [view={40}{60},
    hide axis]
    ]

    domain=-2:2,
    samples=100,
    samples y=2,
]

\addplot3[surf,
	domain=-1:1,
	samples=100,
	samples y=1,
]
({-2*x^3},{x^4},{3*x^2});
\end{axis}
\end{tikzpicture}
